                    <script>
                        var number = 2;
                       
                        document.write(number, " to the power of 5 = ", power(number, 5), "<br>");
                        
                        number = 8;
                        
                        document.write(number, " to the power of 2 = ", power(number), "<br>");
                        
                        function power(number, power = 2)
                        {
                            var result = 1;
                            
                            for(var index = 0; index < power; ++index)
                            {
                                result *= number;
                            }
                            
                            return result;
                        }
                    </script>

                        <input type="button" value="Oblicz"
                        onclick="obliczSilnie(document.myForm.myNumber.value)">

                    <p id="result"></p>
                
                    <script>
                
                        function obliczSilnie(liczba)
                        {
                            var wynik = silnia(liczba);
                
                            if (wynik > 0)
                            {  
                                document.getElementById("result").innerHTML =
                                    "<h3 style='color:blue;'>Silnia liczby " + liczba + " wynosi: " + wynik + "</h3>";
                            }
                        }
                
                        function silnia(liczba)
                        {
                            if (liczba == 0 || liczba == 1)
                            {
                                return 1;
                            }
                            else if (liczba < 0)
                            {  
                                error();
                                return -1;
                            }
                
                            var wynik = liczba * silnia(liczba - 1);
                
                            return wynik;
                        }
                
                        function error()
                        {
                            document.getElementById("result").innerHTML = "<h3 style='color:red;'> Błąd: Nie można obliczyć silni dla liczby ujemnej. </h3>";
                        }    
                
                    </script>
